```rust
/// Parameters for the `merge_check` tool.
#[derive(Deserialize, JsonSchema)]
pub struct MergeCheckParams {
    /// The base ref to merge into (e.g., 'main', 'origin/main', a SHA, or relative ref like 'HEAD~3').
    #[schemars(description = "Base ref (the branch being merged into). Accepts branches, tags, SHAs, or relative refs.")]
    pub base: String,

    /// The head ref to merge from (e.g., 'feature/auth', a SHA, or relative ref).
    #[schemars(description = "Head ref (the branch being merged). Accepts branches, tags, SHAs, or relative refs.")]
    pub head: String,

    /// Maximum number of conflicts to return. Default: 20. Remaining conflicts are counted but not detailed.
    #[schemars(description = "Maximum conflicts to return in detail (default: 20). Excess conflicts are truncated with a count.")]
    #[serde(default)]
    pub max_conflicts: Option<u32>,
}
```

**Tool method** — Add to the `#[tool_router] impl AssayServer` block, after the worktree tools and before the session tools:

```rust
/// Check for merge conflicts between two refs using git merge-tree (read-only, zero side effects).
#[tool(
    description = "Check for merge conflicts between two git refs. Uses `git merge-tree --write-tree` — read-only with zero side effects (no index mutation, no working tree changes). Returns MergeCheck with clean/conflicts status, file changes, ahead/behind counts, and fast-forward detection. Does not require an Assay project — works in any git repository."
)]
pub async fn merge_check(
    &self,
    params: Parameters<MergeCheckParams>,
) -> Result<CallToolResult, McpError> {
    let cwd = resolve_cwd()?;

    let result = match assay_core::merge::merge_check(
        &cwd,
        &params.0.base,
        &params.0.head,
        params.0.max_conflicts,
    ) {
        Ok(check) => check,
        Err(e) => return Ok(domain_error(&e)),
    };

    let json = serde_json::to_string(&result)
        .map_err(|e| McpError::internal_error(format!("serialization failed: {e}"), None))?;
    Ok(CallToolResult::success(vec![Content::text(json)]))
}
```

Note: This tool does NOT call `load_config()` — merge check is a pure git operation that works in any git repo, even without `.assay/` initialization. This matches the CONTEXT decision "Pure git operation: No spec_id parameter, no Assay-specific awareness."
