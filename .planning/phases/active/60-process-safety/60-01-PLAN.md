<!-- Current sanitize function at app.rs:320-334 — to be replaced -->
fn sanitize(s: String) -> String {
    if s.bytes().any(|b| b < 0x20 && b != b'\t') {
        s.chars().map(|c| {
            if (c as u32) < 0x20 && c != '\t' { '\u{FFFD}' } else { c }
        }).collect()
    } else { s }
}
