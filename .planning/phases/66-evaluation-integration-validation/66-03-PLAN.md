   ```rust
   if let Some(extends) = &gates.extends {
       if let Err(e) = compose::validate_slug(extends) {
           diagnostics.push(Diagnostic { severity: Severity::Error, location: "extends".into(), message: format!("invalid slug `{extends}`: {}", extract_slug_reason(&e)) });
       } else {
           // Slug is valid — check existence and cycles (SAFE-01)
       }
   }
   ```

   ```rust
   if let Some(includes) = &gates.include {
       for (i, slug) in includes.iter().enumerate() {
           if let Err(e) = compose::validate_slug(slug) {
               diagnostics.push(Diagnostic { severity: Severity::Error, location: format!("include[{i}]"), message: format!("invalid slug `{slug}`: {}", extract_slug_reason(&e)) });
           }
       }
   }
   ```

7. **Empty includes warning:**
   If `gates.include` is `Some(vec)` and `vec.is_empty()`, emit warning: "empty includes list (no-op)".
