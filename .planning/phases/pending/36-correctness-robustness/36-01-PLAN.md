           .map_err(|e| AssayError::io("creating worktree metadata dir", &meta_dir, e))?;
       let meta_path = meta_dir.join("worktree.json");
       let json = serde_json::to_string_pretty(metadata)

10. Rewrite the ahead/behind logic in `status()`. Replace the current `@{upstream}` approach (lines 239-255) with:
    ```rust
    // Read metadata for base branch
    let metadata = read_metadata(worktree_path);
    let base_branch_name = metadata.as_ref().map(|m| m.base_branch.as_str());

    let mut warnings = Vec::new();
    let (ahead, behind) = if let Some(base) = base_branch_name {
        // Try remote-tracking ref first, fall back to local
        let remote_ref = format!("origin/{base}");
        let local_ref = format!("refs/heads/{base}");
        let compare_ref = if git_command(&["rev-parse", "--verify", &format!("refs/remotes/{remote_ref}")], worktree_path).is_ok() {
            remote_ref
        } else if git_command(&["rev-parse", "--verify", &local_ref], worktree_path).is_ok() {
            local_ref
        } else {
            warnings.push(format!("base branch '{base}' not found locally or as remote-tracking ref — ahead/behind unavailable"));
            return Ok(WorktreeStatus {
                spec_slug: spec_slug.to_string(),
                path: worktree_path.to_path_buf(),
                branch,
                head,
                dirty,
                ahead: None,
                behind: None,
                base_branch: Some(base.to_string()),
                warnings,
            });
        };

        git_command(
            &["rev-list", "--left-right", "--count", &format!("HEAD...{compare_ref}")],
            worktree_path,
        )
        .ok()
        .and_then(|output| {
            let parts: Vec<&str> = output.split('\t').collect();
            if parts.len() == 2 {
                Some((
                    Some(parts[0].parse::<usize>().unwrap_or(0)),
                    Some(parts[1].parse::<usize>().unwrap_or(0)),
                ))
            } else {
                None
            }
        })
        .unwrap_or((Some(0), Some(0)))
    } else {
        warnings.push("no worktree metadata found — ahead/behind unavailable".to_string());
        (None, None)
    };
    ```
    Update the final `Ok(WorktreeStatus { ... })` to include `base_branch` and `warnings`.
