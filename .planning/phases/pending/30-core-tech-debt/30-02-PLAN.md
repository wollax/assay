```rust
/// Persist a gate run as a [`GateRunRecord`] with auto-generated run ID.
///
/// Convenience wrapper around [`save()`] that generates the run ID
/// internally and constructs the [`GateRunRecord`]. Callers provide
/// the summary and metadata; ID generation is an implementation detail.
///
/// See [`save()`] for atomicity guarantees and pruning behavior.
pub fn save_run(
    assay_dir: &Path,
    summary: GateRunSummary,
    working_dir: Option<String>,
    max_history: Option<usize>,
) -> Result<SaveResult> {
    let timestamp = chrono::Utc::now();
    let run_id = generate_run_id(&timestamp);
    let record = GateRunRecord {
        run_id,
        assay_version: env!("CARGO_PKG_VERSION").to_string(),
        timestamp,
        working_dir,
        summary,
    };
    save(assay_dir, &record, max_history)
}
```
