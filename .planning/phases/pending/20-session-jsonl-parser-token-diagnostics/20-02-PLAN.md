  use std::io::BufRead;
  use assay_types::context::SessionEntry;

  /// A parsed session entry with its line number and raw byte size.
  pub struct ParsedEntry {
      pub entry: SessionEntry,
      pub line_number: usize,
      pub raw_bytes: usize,
  }

  /// Parse a session JSONL file, tolerating per-line errors.
  /// Returns successfully parsed entries and a count of skipped lines.
  pub fn parse_session(path: &Path) -> crate::Result<(Vec<ParsedEntry>, usize)> {
      let file = File::open(path).map_err(...)?;
      let reader = BufReader::new(file);
      let mut entries = Vec::new();
      let mut skipped = 0;

      for (line_num, line_result) in reader.lines().enumerate() {
          let line = line_result.map_err(...)?;
          if line.trim().is_empty() { continue; }
          match serde_json::from_str::<SessionEntry>(&line) {
              Ok(entry) => entries.push(ParsedEntry {
                  entry,
                  line_number: line_num + 1,
                  raw_bytes: line.len(),
              }),
              Err(_) => skipped += 1,
          }
      }
      Ok((entries, skipped))
  }
  ```

  Key design points:
  - Use `BufReader::lines()` for streaming (not loading whole file)
  - Skip empty lines silently
  - Tolerate parse errors per line (increment skipped counter)
  - Track raw_bytes per entry for bloat size calculation

  /// Analyze a parsed session and produce a DiagnosticsReport.
  pub fn diagnose(path: &Path, session_id: &str) -> crate::Result<DiagnosticsReport> {
      let (entries, _skipped) = parse_session(path)?;
      let file_size = std::fs::metadata(path)?.len();
      let usage = extract_usage(&entries);
      let model = extract_model(&entries);
      let context_window = context_window_for_model(model.as_deref());
      let available = context_window.saturating_sub(SYSTEM_OVERHEAD_TOKENS);
      let context_pct = usage.as_ref().map(|u| (u.context_tokens() as f64 / available as f64) * 100.0);

      let bloat = categorize_bloat(&entries);
      let message_count = entries.iter()
          .filter(|e| matches!(&e.entry, SessionEntry::User(_) | SessionEntry::Assistant(_)))
          .filter(|e| !is_sidechain(&e.entry))
          .count() as u64;

      Ok(DiagnosticsReport {
          session_id: session_id.to_string(),
          file_path: path.to_string_lossy().to_string(),
          file_size_bytes: file_size,
          total_entries: entries.len() as u64,
          message_count,
          model,
          context_window,
          system_overhead: SYSTEM_OVERHEAD_TOKENS,
          usage,
          context_utilization_pct: context_pct,
          bloat,
      })

  fn categorize_bloat(entries: &[ParsedEntry]) -> BloatBreakdown {
      // Track bytes per category
      let mut category_data: HashMap<BloatCategory, (u64, u64)> = HashMap::new(); // (bytes, count)
      let total_bytes: u64 = entries.iter().map(|e| e.raw_bytes as u64).sum();
      let system_reminder_re = Regex::new(r"<system-reminder>").unwrap();

      // Stale read tracker
      let mut read_paths: HashMap<String, Vec<ReadRange>> = HashMap::new();

      for (idx, pe) in entries.iter().enumerate() {
          match &pe.entry {
              SessionEntry::Progress(_) => {
                  add_to(&mut category_data, BloatCategory::ProgressTicks, pe.raw_bytes as u64);
              }
              SessionEntry::FileHistorySnapshot(_) | SessionEntry::QueueOperation(_) |
              SessionEntry::PrLink(_) | SessionEntry::System(_) => {
                  add_to(&mut category_data, BloatCategory::Metadata, pe.raw_bytes as u64);
              }
              SessionEntry::Assistant(a) => {
                  if let Some(msg) = &a.message {
                      for block in &msg.content {
                          match block {
                              ContentBlock::Thinking { thinking } => {
                                  add_to(&mut category_data, BloatCategory::ThinkingBlocks,
                                      thinking.len() as u64);
                              }
                              ContentBlock::Text { text } => {
                                  if system_reminder_re.is_match(text) {
                                      add_to(&mut category_data, BloatCategory::SystemReminders,
                                          text.len() as u64);
                                  }
                              }
                              _ => {}
                          }
                      }
                  }
              }
              SessionEntry::User(u) => {
                  // Check for tool_result content (tool output bloat)
                  // Check for system reminders in user message text
                  // Check for Read tool_use inputs for stale read tracking
                  // (User entries contain tool_result blocks from previous tool_use)
                  if let Some(msg) = &u.message {
                      // Parse content blocks from user message Value
                      // Look for tool_result blocks -> ToolOutput category
                      // Look for text with <system-reminder> -> SystemReminders category
                      // Look for tool_use with name="Read" -> track for stale reads
                      categorize_user_entry(msg, &mut category_data, &system_reminder_re, &mut read_paths, idx);
                  }
              }
              _ => {}
          }
      }

      // Build BloatBreakdown from category_data
      let entries: Vec<BloatEntry> = BloatCategory::all().iter().map(|cat| {
          let (bytes, count) = category_data.get(cat).copied().unwrap_or((0, 0));
          let percentage = if total_bytes > 0 { (bytes as f64 / total_bytes as f64) * 100.0 } else { 0.0 };
          BloatEntry { category: *cat, bytes, count, percentage }
      }).collect();

      BloatBreakdown { entries }
  }
  ```

  The `categorize_user_entry` helper parses `serde_json::Value` for content blocks in user messages, extracting tool_result content sizes (ToolOutput), system reminder text (SystemReminders), and tracking Read tool_use inputs for stale read detection.

  Stale read detection: track `(file_path, offset, limit)` tuples from Read tool_use inputs. A re-read of the same path with overlapping range adds the tool_result content to StaleReads category.

  pub fn list_sessions(
      project_path: Option<&Path>,
      limit: usize,
      include_tokens: bool,
  ) -> crate::Result<Vec<SessionInfo>> {
      let project = project_path
          .map(|p| p.to_path_buf())
          .or_else(|| std::env::current_dir().ok())
          .ok_or_else(|| ...)?;
      let session_dir = find_session_dir(&project)?;
      let mut paths = discover_sessions(&session_dir)?;
      paths.truncate(limit);

      paths.iter().map(|p| {
          let meta = std::fs::metadata(p)?;
          let session_id = p.file_stem().unwrap().to_string_lossy().to_string();
          let token_count = if include_tokens {
              quick_token_estimate(p).ok().flatten().map(|u| u.context_tokens())
          } else { None };
          Ok(SessionInfo { session_id, project: Some(project.to_string_lossy().to_string()), file_path: p.to_string_lossy().to_string(), file_size_bytes: meta.len(), entry_count: 0, last_modified: Some(...), token_count })
      }).collect()
  }
