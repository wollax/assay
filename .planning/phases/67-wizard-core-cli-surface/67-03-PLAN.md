    ```rust
    //! Shared dialoguer prompt helpers for the gate and criteria wizards.
    use anyhow::{Context, Result};
    use assay_types::CriterionInput;

    /// Prompt for a slug via Input::validate_with, rejecting invalid values inline.
    pub(crate) fn prompt_slug(prompt: &str, initial: Option<&str>) -> Result<String> {
        let mut builder = dialoguer::Input::<String>::new().with_prompt(prompt);
        if let Some(init) = initial {
            builder = builder.with_initial_text(init);
        }
        builder
            .validate_with(|input: &String| -> std::result::Result<(), String> {
                assay_core::spec::compose::validate_slug(input.as_str())
                    .map_err(|e| e.to_string())
            })
            .interact_text()
            .context("reading slug")
    }

    /// Inline add-another criterion loop. Used by both gate wizard and criteria new.
    pub(crate) fn prompt_criteria_loop(existing: &[CriterionInput]) -> Result<Vec<CriterionInput>> {
        // If `existing` is non-empty (edit mode), start the list populated and
        // let user add MORE; optionally offer "clear and re-enter". For v1,
        // if existing is non-empty: ask "Keep existing N criteria?" (default yes);
        // if yes, return existing. If no, enter empty and proceed to add loop.
        // Simplest v1 design: always start empty in create mode; in edit mode,
        // the caller decides whether to pre-seed. Keep this helper simple:
        // it always prompts the "Add a criterion?" loop from the current accumulator.
        let mut criteria = existing.to_vec();
        loop {
            let add = dialoguer::Confirm::new()
                .with_prompt("  Add a criterion?")
                .default(criteria.is_empty())
                .interact()
                .context("confirm add criterion")?;
            if !add { break; }
            let name: String = dialoguer::Input::new()
                .with_prompt("    Criterion name")
                .interact_text()?;
            let description: String = dialoguer::Input::new()
                .with_prompt("    Description")
                .allow_empty(true)
                .interact_text()?;
            let cmd_raw: String = dialoguer::Input::new()
                .with_prompt("    Command (Enter to skip)")
                .allow_empty(true)
                .interact_text()?;
            let cmd = if cmd_raw.trim().is_empty() { None } else { Some(cmd_raw.trim().to_string()) };
            criteria.push(CriterionInput { name, description, cmd });
        }
        Ok(criteria)
    }

    /// Generic Select wrapper that returns the chosen item value.
    pub(crate) fn select_from_list(
        prompt: &str,
        items: &[String],
        default_idx: usize,
    ) -> Result<usize> {
        dialoguer::Select::new()
            .with_prompt(prompt)
            .items(items)
            .default(default_idx)
            .interact()
            .context("selecting from list")
    }

    pub(crate) fn multi_select_from_list(
        prompt: &str,
        items: &[String],
        preselected: &[usize],
    ) -> Result<Vec<usize>> {
        let mut builder = dialoguer::MultiSelect::new()
            .with_prompt(prompt)
            .items(items);
        // Mark defaults (checkboxes pre-checked)
        let defaults: Vec<bool> = (0..items.len()).map(|i| preselected.contains(&i)).collect();
        builder = builder.defaults(&defaults);
        builder.interact().context("selecting multiple items")
    }

    #[cfg(test)]

        // Helpers themselves require a TTY to test end-to-end; unit tests here
        // focus on type signatures and non-interactive branches. Visual flow
        // is on the manual-only verification list in VALIDATION.md.
    }
    ```

    ```rust
    pub(crate) fn handle_wizard(edit: Option<String>) -> anyhow::Result<i32> {
        use std::io::IsTerminal;
        if !std::io::stdin().is_terminal() {
            tracing::error!("assay gate wizard requires an interactive terminal.");
            return Ok(1);
        }

        // Resolve paths (same pattern used elsewhere in this command module)
        let root = std::env::current_dir()?;
        let config = assay_core::config::load(&root)?;  // or the existing path-resolution helper
        let assay_dir = config.assay_dir.clone();       // verify field names
        let specs_dir = config.specs_dir.clone();

        // Edit mode: load existing GatesSpec to pre-fill defaults
        let existing: Option<assay_types::GatesSpec> = match &edit {
            Some(slug) => Some(load_gate_for_edit(slug, &specs_dir)?),
            None => None,
        };

        // Prompt sequence (linear, no back-nav)
        let slug = match &edit {
            Some(slug) => slug.clone(),  // slug fixed in edit mode
            None => super::wizard_helpers::prompt_slug(
                "Gate name (slug)",
                None,
            )?,
        };

        let description: String = {
            let mut b = dialoguer::Input::<String>::new()
                .with_prompt("Description")
                .allow_empty(true);
            if let Some(ref g) = existing {
                if let Some(ref d) = g.description {  // verify field type
                    b = b.with_initial_text(d);
                }
            }
            b.interact_text()?
        };

        // Extends: Select from scanned gates + "(none)" option
        let scan = assay_core::spec::scan(&specs_dir)?;
        let mut gate_options: Vec<String> = scan.entries.iter()
            .map(|e| e.slug().to_string())
            .filter(|s| edit.as_ref().map(|edit_slug| s != edit_slug).unwrap_or(true))  // exclude self
            .collect();
        gate_options.insert(0, "(none)".to_string());
        let default_extends_idx = existing
            .as_ref()
            .and_then(|g| g.extends.as_ref())
            .and_then(|e| gate_options.iter().position(|o| o == e))
            .unwrap_or(0);
        let extends_idx = super::wizard_helpers::select_from_list(
            "Extends (parent gate)", &gate_options, default_extends_idx,
        )?;
        let extends = if extends_idx == 0 { None } else { Some(gate_options[extends_idx].clone()) };

        // Include: MultiSelect of scanned libraries
        let libs = assay_core::spec::compose::scan_libraries(&assay_dir)?;
        let lib_names: Vec<String> = libs.iter().map(|l| l.name.clone()).collect();
        let preselected: Vec<usize> = existing
            .as_ref()
            .map(|g| &g.include)
            .map(|inc| inc.iter().filter_map(|name| lib_names.iter().position(|n| n == name)).collect())
            .unwrap_or_default();
        let include_indices = super::wizard_helpers::multi_select_from_list(
            "Include criteria libraries", &lib_names, &preselected,
        )?;
        let include: Vec<String> = include_indices.iter().map(|&i| lib_names[i].clone()).collect();

        // Criteria: inline loop. In edit mode, offer to keep existing.
        let existing_criteria: Vec<assay_types::CriterionInput> = existing
            .as_ref()
            .map(|g| g.criteria.iter().map(|c| assay_types::CriterionInput {
                name: c.name.clone(),
                description: c.description.clone(),
                cmd: c.cmd.clone(),  // verify the Criterion field name for command
            }).collect())
            .unwrap_or_default();
        let criteria = super::wizard_helpers::prompt_criteria_loop(&existing_criteria)?;

        // Preconditions: opt-in
        let add_preconditions = dialoguer::Confirm::new()
            .with_prompt("Add preconditions?")
            .default(existing.as_ref().map(|g| g.preconditions.is_some()).unwrap_or(false))
            .interact()?;
        let preconditions = if add_preconditions {
            Some(prompt_preconditions(&specs_dir, existing.as_ref().and_then(|g| g.preconditions.as_ref()))?)
        } else {
            None
        };

        // Final confirm
        let write = dialoguer::Confirm::new()
            .with_prompt("Write gate?")
            .default(true)
            .interact()?;
        if !write {
            println!("  aborted (no file written)");
            return Ok(0);
        }

        let input = assay_types::GateWizardInput {
            slug: slug.clone(),
            description: if description.is_empty() { None } else { Some(description) },
            extends,
            include,
            criteria,
            preconditions,
            overwrite: edit.is_some(),
        };
        let output = assay_core::wizard::apply_gate_wizard(&input, &assay_dir, &specs_dir)?;
        println!("  {} gate '{}'", if edit.is_some() { "Updated" } else { "Created" }, slug);
        println!("    written {}", output.path.display());
        Ok(0)
    }

    fn load_gate_for_edit(slug: &str, specs_dir: &std::path::Path) -> anyhow::Result<assay_types::GatesSpec> {
        use assay_core::spec::SpecEntry;
        let entry = assay_core::spec::load_spec_entry_with_diagnostics(slug, specs_dir)
            // enriched_error_display already applied by the diagnostics variant; anyhow
            // will carry the formatted message to the CLI error printer.
            .map_err(anyhow::Error::from)?;
        match entry {
            SpecEntry::Directory { gates, .. } => Ok(gates),
            SpecEntry::Legacy { .. } => anyhow::bail!(
                "gate '{}' uses the legacy flat-file format; gate wizard only supports directory-based specs",
                slug
            ),
        }
    }

    fn prompt_preconditions(
        specs_dir: &std::path::Path,
        existing: Option<&assay_types::SpecPreconditions>,
    ) -> anyhow::Result<assay_types::SpecPreconditions> {
        // requires: MultiSelect of all spec slugs from scan
        let scan = assay_core::spec::scan(specs_dir)?;
        let slugs: Vec<String> = scan.entries.iter().map(|e| e.slug().to_string()).collect();
        let preselected: Vec<usize> = existing
            .map(|p| p.requires.iter().filter_map(|r| slugs.iter().position(|s| s == r)).collect())
            .unwrap_or_default();
        let requires_indices = super::wizard_helpers::multi_select_from_list(
            "Requires specs", &slugs, &preselected,
        )?;
        let requires = requires_indices.iter().map(|&i| slugs[i].clone()).collect();

        // commands: add-another loop
        let mut commands: Vec<String> = existing.map(|p| p.commands.clone()).unwrap_or_default();
        loop {
            let add = dialoguer::Confirm::new()
                .with_prompt("  Add a precondition command?")
                .default(commands.is_empty())
                .interact()?;
            if !add { break; }
            let cmd: String = dialoguer::Input::new()
                .with_prompt("    Command")
                .interact_text()?;
            commands.push(cmd);
        }

        Ok(assay_types::SpecPreconditions { requires, commands })
    }
    ```
