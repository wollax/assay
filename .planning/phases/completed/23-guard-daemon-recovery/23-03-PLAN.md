//! File system watcher for reactive session monitoring.
//!
//! Wraps `notify::RecommendedWatcher` to detect session file growth
//! with sub-second latency (kqueue on macOS, inotify on Linux).

use std::path::{Path, PathBuf};
use std::time::{Duration, Instant};

use notify::{Config, Event, EventKind, RecommendedWatcher, RecursiveMode, Watcher};
use tokio::sync::mpsc;

/// Watches a session file for modifications.
pub struct SessionWatcher {
    /// The watcher must be kept alive for events to flow.
    _watcher: RecommendedWatcher,
    /// Channel receiving filtered events.
    pub rx: mpsc::UnboundedReceiver<()>,
    /// Target session file path for filtering.
    target: PathBuf,
}
```

2. Implement SessionWatcher:
   - `pub fn new(session_path: &Path) -> crate::Result<Self>`:
     - Create a `tokio::sync::mpsc::unbounded_channel::<()>`
     - Create `RecommendedWatcher` with an event handler closure that:
       - Filters events: only `EventKind::Modify(_)` or `EventKind::Create(_)`
       - Checks that the event's paths include the target session file (compare canonical paths)
       - Sends `()` on the channel for matching events
     - Watch the session file's parent directory (not the file itself, since atomic writes create new files). Use `RecursiveMode::NonRecursive`.
     - Configure with `Config::default().with_poll_interval(Duration::from_millis(500))` as fallback poll interval
     - Store the watcher in the struct to prevent it from being dropped
   - Event filtering is critical: ignore events for temp files (e.g., `.tmp`, `~` suffixes) created during atomic writes. Only pass through events where at least one path matches the session file name.

3. Add `pub mod watcher;` to guard/mod.rs.

4. Write tests:
   - Test that SessionWatcher can be created for a valid path (constructor doesn't panic)
   - Test that modifying the watched file sends an event on the channel
   - Test that modifying a different file in the same directory does NOT send an event

//! Guard daemon main event loop.
//!
//! Multiplexes polling timer, file system watcher, and shutdown signals.
//! Responds to threshold crossings with escalating pruning prescriptions.

use std::path::{Path, PathBuf};
use std::time::Duration;

use assay_types::GuardConfig;
use tokio::signal;
use tracing::{info, warn, error};

use super::circuit_breaker::CircuitBreaker;
use super::pid;
use super::thresholds::{ThresholdLevel, evaluate_thresholds};
use super::watcher::SessionWatcher;

/// Guard daemon state.
pub struct GuardDaemon {
    session_path: PathBuf,
    assay_dir: PathBuf,
    config: GuardConfig,
    breaker: CircuitBreaker,
}
```

2. Implement GuardDaemon methods:

   - `pub fn new(session_path: PathBuf, assay_dir: PathBuf, config: GuardConfig) -> Self`:
     Create CircuitBreaker from config fields.

   - `pub async fn run(&mut self) -> crate::Result<()>`:
     Main event loop (NOTE: tokio must be a dep of assay-core; verify it is in Cargo.toml or add it):
     ```
     a. Create PID file (fail with GuardAlreadyRunning if exists)
     b. Set up SessionWatcher on session_path
     c. Set up poll interval timer: tokio::time::interval(Duration::from_secs(config.poll_interval_secs))
     d. Set up signal handlers using tokio::signal::unix::signal:
        - SIGINT (Ctrl+C): signal(SignalKind::interrupt())
        - SIGTERM (from stop_guard): signal(SignalKind::terminate())
     e. Loop with tokio::select!:
        - poll_tick => check_and_respond()
        - watcher.rx.recv() => debounce (skip if <1s since last check), then check_and_respond()
        - sigint.recv() => graceful_shutdown(), break
        - sigterm.recv() => graceful_shutdown(), break
     f. After loop: remove PID file
     ```

   - `async fn check_and_respond(&mut self) -> crate::Result<bool>`:
     ```
     a. Get file size: std::fs::metadata(session_path).map(|m| m.len())
     b. Get token estimate: context::quick_token_estimate(session_path).
        Calculate context_pct from token estimate.
     c. Call evaluate_thresholds(config, context_pct, file_size)
     d. Match threshold level:
        - None => breaker.reset_if_quiet(), return Ok(false)
        - Soft => handle_soft_threshold()
        - Hard => handle_hard_threshold()
     e. After prune: re-evaluate thresholds (P4 pitfall: stale state after prune)
     f. Return Ok(true) if action was taken
     ```

   - `async fn handle_soft_threshold(&mut self) -> crate::Result<()>`:
     ```
     a. Record recovery in circuit breaker (returns count)
     b. Check should_trip(); if true: trip it, save final checkpoint, return GuardCircuitBreakerTripped error
     c. Get current tier from breaker (escalating prescription)
     d. Save checkpoint before pruning
     e. Run prune_session with tier.strategies(), execute=true
     f. Log: "[guard] Soft threshold — {tier} prune saved {bytes} bytes"
     ```

   - `async fn handle_hard_threshold(&mut self) -> crate::Result<()>`:
     ```
     a. Record recovery in circuit breaker (returns count)
     b. Check should_trip(); if true: trip it, save final checkpoint, return GuardCircuitBreakerTripped error
     c. Get current tier from breaker (always at least Standard for hard threshold; use max(breaker.current_tier(), Standard))
     d. Save checkpoint (team-protect) before pruning
     e. Run prune_session with tier.strategies(), execute=true
     f. Log: "[guard] Hard threshold — {tier} prune saved {bytes} bytes"
     g. Log recommendation: "Consider running /compact in your Claude Code session"
     ```

   - `fn graceful_shutdown(&self) -> crate::Result<()>`:
     ```
     a. Log: "[guard] Shutting down..."
     b. Save final checkpoint
     c. Remove PID file
     d. Log: "[guard] Final checkpoint saved. Exiting."
     ```

   Important implementation notes:
   - Debounce watcher events: track `last_check: Instant` and skip if elapsed < 1 second (P5 pitfall)
   - The watcher must NOT be dropped during the loop (P1 pitfall)
   - Token estimate may fail (file being written); treat errors as "no data, skip this tick"
   - All checkpoint/prune operations should log errors but not crash the daemon (continue loop)
   - Circuit breaker trip IS a terminal condition (exit the loop with error)

3. Update `crates/assay-core/src/guard/mod.rs` to add public API functions:

   ```rust
   pub mod daemon;
   pub mod watcher;

   /// Start the guard daemon. Blocks until shutdown or circuit breaker trip.
   pub async fn start_guard(
       session_path: &Path,
       assay_dir: &Path,
       config: GuardConfig,
   ) -> crate::Result<()> {
       let mut daemon = daemon::GuardDaemon::new(
           session_path.to_path_buf(),
           assay_dir.to_path_buf(),
           config,
       );
       daemon.run().await
   }

   /// Stop a running guard daemon by reading its PID file and sending SIGTERM.
   pub fn stop_guard(assay_dir: &Path) -> crate::Result<()> {
       let pid_path = pid::pid_file_path(assay_dir);
       match pid::check_running(&pid_path) {
           Some(pid) => {
               unsafe { libc::kill(pid as i32, libc::SIGTERM); }
               // Wait briefly for process to exit, then clean up PID file
               std::thread::sleep(std::time::Duration::from_millis(500));
               pid::remove_pid_file(&pid_path)?;
               Ok(())
           }
           None => Err(crate::AssayError::GuardNotRunning),
       }
   }

   /// Check guard daemon status.
   pub fn guard_status(assay_dir: &Path) -> GuardStatus {
       let pid_path = pid::pid_file_path(assay_dir);
       match pid::check_running(&pid_path) {
           Some(pid) => GuardStatus::Running { pid },
           None => GuardStatus::Stopped,
       }
   }

   /// Guard daemon status.
   #[derive(Debug)]
   pub enum GuardStatus {
       Running { pid: u32 },
       Stopped,
   }
   ```
