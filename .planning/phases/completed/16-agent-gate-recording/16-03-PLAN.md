2. Implement `gate_finalize` tool:
   ```rust
   #[tool(description = "Finalize an active gate session, persisting all accumulated evaluations as a GateRunRecord. Missing required criteria cause the gate to fail. Call this after submitting all gate_report evaluations.")]
   async fn gate_finalize(&self, params: Parameters<GateFinalizeParams>) -> Result<CallToolResult, McpError> {
       let session_id = params.0.session_id;
       // 1. Remove session from map (atomic — prevents timeout race)
       let session = {
           let mut sessions = self.sessions.lock().await;
           sessions.remove(&session_id)
       };
       let Some(session) = session else {
           return Ok(CallToolResult::error(vec![Content::text(
               format!("session '{}' not found", session_id)
           )]));
       };
       // 2. Resolve assay_dir and config
       let cwd = resolve_cwd()?;
       let config = match load_config(&cwd) { Ok(c) => c, Err(e) => return Ok(e) };
       let assay_dir = cwd.join(".assay");
       let working_dir = resolve_working_dir(&cwd, &config);
       let max_history = config.gates.as_ref().and_then(|g| g.max_history);
       // 3. Finalize (no lock held — safe for I/O)
       let record = match assay_core::gate::session::finalize_session(
           session, &assay_dir, Some(working_dir.to_string_lossy().to_string()), max_history
       ) {
           Ok(r) => r,
           Err(e) => return Ok(domain_error(&e)),
       };
       // 4. Return summary response
       let response = serde_json::json!({
           "run_id": record.run_id,
           "spec_name": record.summary.spec_name,
           "passed": record.summary.passed,
           "failed": record.summary.failed,
           "skipped": record.summary.skipped,
           "required_failed": record.summary.enforcement.required_failed,
           "advisory_failed": record.summary.enforcement.advisory_failed,
           "persisted": true,
       });
       Ok(CallToolResult::success(vec![Content::text(serde_json::to_string(&response).unwrap())]))
   }
   ```

3. Extend `gate_run` to auto-create sessions when spec contains AgentReport criteria:
   - After evaluating command criteria via spawn_blocking, check if any criteria have `kind == Some(CriterionKind::AgentReport)`
   - If yes: create a session with `assay_core::gate::session::create_session`, passing `summary.results` as the `command_results` parameter (these are the already-evaluated non-agent criteria results from the gate_run). Store it in self.sessions.
   - Spawn a timeout task, capturing `assay_dir` and `max_history` from the current gate_run context (do NOT use `std::env::current_dir()` at timeout time — CWD may have changed):
     ```rust
     let sessions_clone = self.sessions.clone();
     let sid = session.session_id.clone();
     let timeout_assay_dir = assay_dir.clone();
     let timeout_max_history = max_history;
     tokio::spawn(async move {
         tokio::time::sleep(Duration::from_secs(SESSION_TIMEOUT_SECS)).await;
         let session = {
             let mut sessions = sessions_clone.lock().await;
             sessions.remove(&sid)
         };
         if let Some(session) = session {
             let record = assay_core::gate::session::finalize_as_timed_out(session);
             // Best-effort persist using the assay_dir captured at session creation
             let _ = assay_core::history::save(&timeout_assay_dir, &record, timeout_max_history);
         }
     });
     ```

     ```rust
     session_id: Option<String>,
     #[serde(skip_serializing_if = "Option::is_none")]
     pending_criteria: Option<Vec<String>>,
     ```
   - If no agent criteria, gate_run behaves exactly as before (no session).
