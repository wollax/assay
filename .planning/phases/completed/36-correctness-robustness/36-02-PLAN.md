5. Create a shared helper method on `AssayServer` (or as a free function):
   ```rust
   /// Build a session-not-found error with context-aware recovery hints.
   async fn session_not_found_error(
       session_id: &str,
       timed_out_sessions: &Mutex<HashMap<String, TimedOutInfo>>,
   ) -> CallToolResult {
       let timed_out = timed_out_sessions.lock().await;
       if let Some(info) = timed_out.get(session_id) {
           let elapsed = (info.timed_out_at - info.created_at).num_seconds();
           CallToolResult::error(vec![Content::text(format!(
               "Session '{session_id}' timed out after {elapsed}s (timeout: {}s). \
                Use gate_run to start a new session.",
               info.timeout_secs
           ))])
       } else {
           CallToolResult::error(vec![Content::text(format!(
               "Session '{session_id}' not found (expired or already finalized). \
                Use gate_run to start a new session, or gate_history to check past results.",
           ))])
       }
   }
   ```
