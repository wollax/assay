3. Add the tool handler method to the `#[tool_handler]` impl block, following the pattern of `spec_get`. Place it after `spec_get` and before `gate_run`:

```rust
#[tool(
    description = "Statically validate a spec without running it. Returns a ValidationResult with \
        per-criterion diagnostics, severity levels (error/warning/info), and a summary. \
        Errors block validity (TOML parse errors, duplicate criterion names, missing required fields). \
        Warnings are advisory (missing AgentReport prompt, command not found on PATH). \
        When the spec declares depends=[...], cross-spec dependency cycles are detected. \
        Use check_commands=true to verify command binaries exist on PATH."
)]
pub async fn spec_validate(
    &self,
    #[tool(aggr)] params: SpecValidateParams,
) -> Result<CallToolResult, McpError> {
    let specs_dir = self.specs_dir().await.map_err(|e| {
        McpError::internal_error(e.to_string(), None)
    })?;

    // First, try to load the spec entry (this handles TOML parse errors)
    let entry = match assay_core::spec::load_spec_entry_with_diagnostics(
        &params.name,
        &specs_dir,
    ) {
        Ok(entry) => entry,
        Err(assay_core::AssayError::SpecParse { path, message })
        | Err(assay_core::AssayError::GatesSpecParse { path, message }) => {
            // TOML parse error ‚Äî return as ValidationResult with error diagnostic
            let result = assay_types::ValidationResult {
                spec: params.name.clone(),
                valid: false,
                diagnostics: vec![assay_types::Diagnostic {
                    severity: assay_types::Severity::Error,
                    location: "toml".to_string(),
                    message,
                }],
                summary: assay_types::DiagnosticSummary {
                    errors: 1,
                    warnings: 0,
                    info: 0,
                },
            };
            return Ok(CallToolResult::success(vec![Content::json(result)?]));
        }
        Err(assay_core::AssayError::SpecValidation { path, errors }) => {
            // Validation errors from load ‚Äî convert to ValidationResult
            let diagnostics: Vec<assay_types::Diagnostic> = errors
                .iter()
                .map(|e| assay_types::Diagnostic {
                    severity: assay_types::Severity::Error,
                    location: e.field.clone(),
                    message: e.message.clone(),
                })
                .collect();
            let error_count = diagnostics.len();
            let result = assay_types::ValidationResult {
                spec: params.name.clone(),
                valid: false,
                diagnostics,
                summary: assay_types::DiagnosticSummary {
                    errors: error_count,
                    warnings: 0,
                    info: 0,
                },
            };
            return Ok(CallToolResult::success(vec![Content::json(result)?]));
        }
        Err(assay_core::AssayError::GatesSpecValidation { path, errors }) => {
            let diagnostics: Vec<assay_types::Diagnostic> = errors
                .iter()
                .map(|e| assay_types::Diagnostic {
                    severity: assay_types::Severity::Error,
                    location: e.field.clone(),
                    message: e.message.clone(),
                })
                .collect();
            let error_count = diagnostics.len();
            let result = assay_types::ValidationResult {
                spec: params.name.clone(),
                valid: false,
                diagnostics,
                summary: assay_types::DiagnosticSummary {
                    errors: error_count,
                    warnings: 0,
                    info: 0,
                },
            };
            return Ok(CallToolResult::success(vec![Content::json(result)?]));
        }
        Err(other) => {
            return Ok(domain_error(&other));
        }
    };

    // Spec loaded successfully ‚Äî run validation with additional checks
    let result = assay_core::spec::validate::validate_spec_with_dependencies(
        &entry,
        params.check_commands,
        &specs_dir,
    );

    Ok(CallToolResult::success(vec![Content::json(result)?]))
}
```

IMPORTANT NOTES:
- The `load_spec_entry_with_diagnostics` function already calls `load()` which calls `validate()`. If it returns Ok, the spec passed core validation. The `validate_spec_with_dependencies` call then adds additional checks (agent prompts, commands, cycles).
- When load fails with parse/validation errors, we catch those and convert to ValidationResult directly (rather than returning a domain error). This ensures the tool ALWAYS returns structured ValidationResult JSON.
- The `SpecNotFound` and `SpecNotFoundDiagnostic` errors are returned via `domain_error()` as-is (not as ValidationResult) since the spec doesn't exist.
