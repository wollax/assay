Replace the byte-budget diff truncation in gate_evaluate's Step 3 with token-aware budgeting. This is a localized change to the gate_evaluate handler in server.rs.

**1a. Remove or deprecate `DIFF_BUDGET_BYTES`:**

The `DIFF_BUDGET_BYTES = 32 * 1024` constant is no longer needed for gate_evaluate. Check if it's used elsewhere (gate_run handler also uses it). If gate_run still uses it, leave the constant but remove the reference from gate_evaluate. If gate_evaluate is the only user, remove it.

**1b. Restructure Step 3 to use budget_context:**

The current flow is:
1. Run `git diff HEAD`
2. Truncate with `truncate_diff(&raw, DIFF_BUDGET_BYTES)`
3. Pass truncated diff to `build_evaluator_prompt`

The new flow reorders steps slightly — budget_context needs the system prompt, spec body, and criteria text to compute overhead. So Steps 3-5 are restructured:

```
Step 3: Capture raw git diff (no truncation yet)
Step 4: Build system prompt + criteria text (moved before budgeting)
Step 5: Call budget_context to get token-budgeted diff
Step 6: Build evaluator prompt with (possibly truncated) diff
Step 7+: remainder unchanged
```

**Detailed implementation for the new Step 3-6:**

```rust
// ── Step 3: Capture raw git diff ──────────────────────────────────
let raw_diff = {
    match std::process::Command::new("git")
        .args(["diff", "HEAD"])
        .current_dir(&working_dir)
        .output()
    {
        Ok(output) if output.status.success() => {
            let raw = String::from_utf8_lossy(&output.stdout).into_owned();
            if raw.is_empty() { None } else { Some(raw) }
        }
        Ok(output) => {
            let stderr = String::from_utf8_lossy(&output.stderr);
            let msg = format!("git diff HEAD failed: {}", stderr.trim());
            tracing::warn!("{msg}");
            warnings.push(format!("{msg} — evaluator will assess without diff context"));
            None
        }
        Err(e) => {
            let msg = format!("git diff HEAD command error: {e}");
            tracing::warn!("{msg}");
            warnings.push(format!("{msg} — evaluator will assess without diff context"));
            None
        }
    }
};

// ── Step 4: Build system prompt and schema (moved before budgeting) ──
let system_prompt = assay_core::evaluator::build_system_prompt();
let schema_json = assay_core::evaluator::evaluator_schema_json();

// Build criteria text for budget computation.
// build_evaluator_prompt also builds criteria text internally — accept the
// double computation (cheap for typical criterion counts, avoids refactoring
// build_evaluator_prompt's API). See research Pitfall 3.
let criteria_text: String = criteria
    .iter()
    .enumerate()
    .map(|(i, c)| {
        let mut s = format!("### {}. {}", i + 1, c.name);
        if let Some(desc) = &c.description {
            s.push_str(&format!("\n{desc}"));
        }
        if let Some(prompt) = &c.prompt {
            s.push_str(&format!("\nEvaluation guidance: {prompt}"));
        }
        s
    })
    .collect::<Vec<_>>()
    .join("\n\n");

// ── Step 5: Token-budget the diff via budget_context ──────────────
let model_window = assay_core::context::context_window_for_model(Some(&model));

let (diff, diff_truncation) = if let Some(ref raw) = raw_diff {
    match assay_core::context::budget_context(
        &system_prompt,
        &description,
        &criteria_text,
        raw,
        model_window,
    ) {
        Ok(budgeted) => {
            // Extract the (possibly truncated) diff from budget_context output.
            // Canonical order: [system_prompt, spec_body, criteria_text, diff]
            // Diff is last if present and non-empty.
            let truncated_diff = budgeted.last().cloned().filter(|s| {
                // Distinguish diff from criteria_text: diff starts with "diff --git"
                // or could be truncated. Compare by length against raw.
                s != &system_prompt && s != &description && s != &criteria_text
            });

            let was_truncated = match &truncated_diff {
                Some(d) => d.len() < raw.len(),
                None => !raw.is_empty(), // diff was fully dropped
            };

            if was_truncated {
                let raw_files = assay_core::gate::extract_diff_files(raw);
                let kept_files = truncated_diff
                    .as_deref()
                    .map(assay_core::gate::extract_diff_files)
                    .unwrap_or_default();
                let omitted_files: Vec<String> = raw_files
                    .iter()
                    .filter(|f| !kept_files.contains(f))
                    .cloned()
                    .collect();

                let original_bytes = raw.len();
                let truncated_bytes = truncated_diff.as_ref().map(|d| d.len()).unwrap_or(0);
                let omitted_count = omitted_files.len();

                warnings.push(format!(
                    "Diff truncated from {original_bytes} to {truncated_bytes} bytes \
                     ({} files kept, {omitted_count} files omitted) to fit token budget",
                    kept_files.len()
                ));

                let meta = assay_types::DiffTruncation {
                    original_bytes,
                    truncated_bytes,
                    included_files: kept_files,
                    omitted_files,
                };
                (truncated_diff, Some(meta))
            } else {
                (truncated_diff.or_else(|| Some(raw.clone())), None)
            }
        }
        Err(e) => {
            tracing::warn!("budget_context failed: {e}");
            warnings.push(format!(
                "Diff budget computation failed: {e} — full diff passed to evaluator"
            ));
            (Some(raw.clone()), None)
        }
    }
} else {
    (None, None)
};

// ── Step 6: Build evaluator prompt ────────────────────────────────
let prompt = assay_core::evaluator::build_evaluator_prompt(
    &spec_name,
    &description,
    &criteria,
    diff.as_deref(),
    None,
);
```
